- **Correction 2 (opérateur) :** `fit_dmd` PUIS `clip_eigenvalues(A, 1.0)` — le mouillé/sec produit plus facilement des tail modes |λ|>1 ; un blow-up de DMD brut serait un FAUX verdict « opérateur casse ». Toujours écrêter.

- **Critère de succès = VISUEL** (plausibilité du rendu), pas une L2 d'oracle. Pas de critère L2 d'acceptation.
- Solveur 2e ordre validé comme générateur (ne pas re-valider) ; `solver.py`/1er ordre intacts. numpy/scipy/matplotlib. Outputs/ gitignoré.
- **C-G/Synolakis = débogage contingent** (si run-up rendu paraît faux) — PAS dans ce plan.
- **Remède orthogonal** : base explose → encodeur de base ; opérateur casse visuellement → dynamique plus riche (PAS forcément encodeur). Ne pas confondre les deux verdicts.

def test_moving_wave_propagates_one_way_not_split():
    # Correction 1 : avec u co-directionnel, l'onde va vers +x ; SANS, elle se scinde.
    b = slope_bed(GRID, slope=0.3, axis="x")        # plage montante vers +x
    h0, hu0, hv0 = moving_wave_ic(GRID, b, still_surface=0.0, amp=0.15,

    _, hs, _, _ = simulate_wetdry_o2(h0, hu0, hv0, b, GRID, t_end=0.3)

    c0 = (xs * pert0).sum() / (pert0.sum() + 1e-12)
    cT = (xs * pertT).sum() / (pertT.sum() + 1e-12)
    assert cT > c0 + GRID.dx                          # le paquet a avancé (pas scindé/immobile)

```python
"""Vocabulaire de transport mouillé/sec : CI (η + u co-directionnel) + scénarios.
PESÉ run-up (seul régime exerçant l'assèchement h>0->0, le cas opérateur le plus dur)
et positions étalées (axe V5 = position autant qu'orientation). Généré par le solveur
2e ordre validé. Sert deux verdicts : étendue (k combiné) + opérateur single-global."""
from __future__ import annotations
from dataclasses import dataclass
import numpy as np
from config import GridConfig, GRAVITY
from src.terrains import gaussian_terrain

_DIRS = {"+x": (1.0, 0.0), "-x": (-1.0, 0.0), "+y": (0.0, 1.0), "-y": (0.0, -1.0),
         "diag": (np.sqrt(0.5), np.sqrt(0.5))}


def slope_bed(grid: GridConfig, slope: float, axis: str = "x") -> np.ndarray:
    """Plage inclinée : b monte selon l'axe (centrée pour traverser le zéro -> mouillé/sec)."""
    H, W = grid.H, grid.W
    yy, xx = np.mgrid[0:H, 0:W].astype(np.float64)
    coord = (xx / (W - 1)) if axis == "x" else (yy / (H - 1))
    return slope * (coord - 0.5)


def island_bed(grid: GridConfig, amp: float, x0_frac: float, y0_frac: float,
               sigma: float) -> np.ndarray:
    """Île émergente : gaussienne dépassant la surface de repos (réutilise gaussian_terrain)."""
    return gaussian_terrain(grid, amp, x0_frac, y0_frac, sigma, slope=0.0)


def moving_wave_ic(grid, b, still_surface, amp, x0_frac, y0_frac, width_frac, direction):
    """Bosse de surface MOBILE : η = amp·sech²(r/width), u co-directionnel = √(g/h)·η.
    Correction 1 : sans u, la bosse se scinde (d'Alembert) -> pas de run-up propre."""
    H, W = grid.H, grid.W
    g = GRAVITY
    yy, xx = np.mgrid[0:H, 0:W].astype(np.float64)
    cx, cy = x0_frac * (W - 1), y0_frac * (H - 1)
    nx, ny = _DIRS[direction]
    # coordonnée le long de la direction de propagation (cellules)
    coord = (xx - cx) * nx + (yy - cy) * ny
    width = width_frac * max(H, W)
    eta_bump = amp / np.cosh(coord / width) ** 2
    h_still = np.maximum(still_surface - b, 0.0)
    wet = h_still > 1e-6
    h = h_still + np.where(wet, eta_bump, 0.0)
    u_mag = np.zeros_like(h)
    u_mag[wet] = np.sqrt(g / h_still[wet]) * eta_bump[wet]   # vitesse linéaire compagnon
    hu = h * (u_mag * nx)
    hv = h * (u_mag * ny)
    return h, hu, hv


def dam_break_ic(grid, b, hl, hr, x0_frac, orientation):
    """Dam-break (sec hr=0 / mouillé hr>0) orienté. h mesuré au-dessus du lit b."""
    H, W = grid.H, grid.W
    yy, xx = np.mgrid[0:H, 0:W].astype(np.float64)
    if orientation == "x":
        coord, c0 = xx / (W - 1), x0_frac
    elif orientation == "y":
        coord, c0 = yy / (H - 1), x0_frac
    else:  # diag
        coord, c0 = 0.5 * (xx / (W - 1) + yy / (H - 1)), x0_frac
    surf = np.where(coord <= c0, hl, hr)
    h = np.maximum(surf - b, 0.0)
    z = np.zeros((H, W))
    return h, z.copy(), z.copy()


@dataclass
class Scenario:
    name: str
    b: np.ndarray
    h0: np.ndarray
    hu0: np.ndarray
    hv0: np.ndarray
    t_end: float


def vocabulary(grid: GridConfig) -> list[Scenario]:
    """~16 scénarios pesés run-up, positions étalées. Amplitude de transport large
    (verdict d'étendue conservateur). Orientations {x,y,diag}."""
    flat = np.zeros((grid.H, grid.W))
    sl_x = slope_bed(grid, slope=0.6, axis="x")
    sl_y = slope_bed(grid, slope=0.6, axis="y")
    isl = island_bed(grid, amp=0.6, x0_frac=0.5, y0_frac=0.5, sigma=8.0)
    isl_off = island_bed(grid, amp=0.6, x0_frac=0.35, y0_frac=0.6, sigma=7.0)
    sc: list[Scenario] = []

    # --- RUN-UP (>=8) : positions x{30,50,70}, terrains pente/île, orientations variées ---
    runup_specs = [
        ("runup_slopex_x30_dxp", sl_x, 0.30, 0.5, "+x"),
        ("runup_slopex_x50_dxp", sl_x, 0.50, 0.5, "+x"),
        ("runup_slopex_x70_dxm", sl_x, 0.70, 0.5, "-x"),
        ("runup_slopey_x40_dyp", sl_y, 0.40, 0.4, "+y"),
        ("runup_slopey_x60_dyp", sl_y, 0.60, 0.6, "+y"),
        ("runup_island_x30_diag", isl, 0.30, 0.5, "diag"),
        ("runup_island_x50_dxp", isl, 0.50, 0.3, "+x"),
        ("runup_islandoff_x60_diag", isl_off, 0.60, 0.5, "diag"),
    ]
    for name, bed, x0f, y0f, d in runup_specs:
        h, hu, hv = moving_wave_ic(grid, bed, still_surface=0.0, amp=0.15,
                                   x0_frac=x0f, y0_frac=y0f, width_frac=0.08, direction=d)
        sc.append(Scenario(name, bed, h, hu, hv, t_end=1.2))

    # --- DAM-BREAK SEC (mouillage) : positions/orientations variées ---
    for name, bed, x0f, ori in [("dambreakdry_x30_x", flat, 0.30, "x"),
                                ("dambreakdry_x50_diag", sl_x, 0.50, "diag"),
                                ("dambreakdry_x60_y", flat, 0.60, "y")]:
        h, hu, hv = dam_break_ic(grid, bed, hl=0.5, hr=0.0, x0_frac=x0f, orientation=ori)
        sc.append(Scenario(name, bed, h, hu, hv, t_end=0.9))

    # --- BORE MOUILLÉ (Stoker) : positions/orientations variées ---
    for name, bed, x0f, ori in [("bore_x40_x", flat, 0.40, "x"),
                                ("bore_x50_diag", flat, 0.50, "diag"),
                                ("bore_x30_y", flat, 0.30, "y")]:
        h, hu, hv = dam_break_ic(grid, bed, hl=1.0, hr=0.1, x0_frac=x0f, orientation=ori)
        sc.append(Scenario(name, bed, h, hu, hv, t_end=0.7))

    return sc
```

```python
def _k_at_error(X_h, all_h, H, W, tau):
    from src.pod import fit_pod, encode, decode, unstack_height
    from src.metrics import relative_l2_error
    basis = fit_pod(X_h, 0.9999999, 4000, n_channels=1)
    for k in range(1, basis.Phi.shape[1] + 1):
        bt = type(basis)(mean=basis.mean, scale=basis.scale,
                         Phi=basis.Phi[:, :k], singular_values=basis.singular_values)
        rec = unstack_height(decode(bt, encode(bt, X_h)), H, W)
        if relative_l2_error(rec, all_h) <= tau:
            return k
    return basis.Phi.shape[1]
```

- `generate(grid)` : pour chaque `Scenario`, `simulate_wetdry_o2(...)` → stocke `h_seq, hu_seq, hv_seq`.
- `extent_endpoint` : empile la hauteur de TOUS les scénarios (`np.concatenate([stack_height(s["h_seq"]) ...], axis=1)`) → `k_combined = _k_at_error(...)`. Calcule aussi `k_per_scenario` médian (par scénario isolé). **Verdict** : `k_combined ≲ 2-3× k_per` → SATURE (étendue bornée) ; `k_combined ≈ Σ k_per` → CROÎT (non bornée → encodeur de base).

def operator_and_rollout(seqs, grid):
    from src.pod import stack_snapshots, fit_pod, encode, decode, unstack
    from src.dmd import fit_dmd, clip_eigenvalues, rollout, spectral_radius
    H, W = grid.H, grid.W
    # POD état complet [h,u,v] sur TOUT le vocabulaire (un seul jeu)
    Xs = [stack_snapshots(s["h_seq"], s["hu_seq"] / np.maximum(s["h_seq"], 1e-6),
                          s["hv_seq"] / np.maximum(s["h_seq"], 1e-6)) for s in seqs]
    Xall = np.concatenate(Xs, axis=1)
    basis = fit_pod(Xall, 0.9999, 2000, n_channels=3)
    z_list = [encode(basis, X) for X in Xs]            # une trajectoire de coeffs par scénario
    A = fit_dmd(z_list)
    rho_raw = spectral_radius(A)
    A = clip_eigenvalues(A, 1.0)                        # CORRECTION 2 : écrêtage |λ|<=1
    rho_clip = spectral_radius(A)
    return basis, A, rho_raw, rho_clip, z_list
